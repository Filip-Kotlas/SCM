71void CRS_Matrix::cg(std::vector<double> &u, std::vector<double> const &f,
72 int const max_iterations, double const eps) const
73{
74 assert(_ncols == _nrows);
75 assert( _ncols == static_cast<int>(u.size()) ); // compatibility of inner dimensions
76 assert( _nrows == static_cast<int>(f.size()) ); // compatibility of outer dimensions
77
78 vector<double> r(size(f));
79 vector<double> w(size(u));
80 vector<double> s(size(u));
81 vector<double> v(size(f));
82 vector<double> D(size(u)); GetDiag(D); // diag(K)
83
84 fill(begin(u),end(u),0.0); // u = 0
85 Defect(r,f,u); // r = f-K*u
86
87 vddiv(w,r,D); // w = D^(-1)*r
88 //w=r;
89 s = w; // s = w
90
91 double sigma = dscapr(w,r); // sig = <w,r>
92 double sigma0(sigma);
93 int iter(0);
94 cout << iter << " iterations : error " << sqrt(sigma) << " rel. error: " << sqrt(sigma/sigma0) << endl;
95
96 while (sigma0*eps*eps<sigma && iter<max_iterations)
97 {
98 ++iter;
99 double sig_old(sigma);
100 Mult(v,s); // v = K*s
101 double alpha = sigma/dscapr(s,v); // alf = sig/<s,v>
102 vdaxpy(u,u,alpha,s); // u = u+alf*s
103 vdaxpy(r,r,-alpha,v); // r = r-alf*v
104 vddiv(w,r,D); // w = D^(-1)*r
105 //w=r;
106 sigma = dscapr(w,r); // sig = <w,r>
107 //cout << iter << " sigma " << sigma << endl;
108 double beta = sigma/sig_old;
109 vdaxpy(s,w,beta,s); // s = w+beta*s
110 }
111 cout << iter << " iterations : error " << sqrt(sigma) << " rel. error: " << sqrt(sigma/sigma0) << endl;
112}

134{
135 int const id2 = _id[row + 1]; // end and

206void CRS_Matrix::getNumberOffDiagonals(int& klow, int& kup) const
207{
208 klow=0; // #subdiagonals
209 kup =0; // #superdiagonals
210 // Assumption: column indices ordered ascending
211 for (int row=0; row<_nrows; ++row)
212 {
213 int const clow = _ik[_id[row]]; // first column index
214 int const cup = _ik[_id[row+1]-1]; // last column index
215 klow = max(klow,row-clow);
216 kup = max( kup,cup-row);
