77void CRS_Matrix::JacobiSmoother(std::vector<double> const &f, std::vector<double> &u,
78 std::vector<double> &r, int nsmooth, double const omega, bool zero) const
79{
80 // ToDO: ensure compatible dimensions
81 assert(_ncols==_nrows);
82 assert( _ncols == static_cast<int>(u.size()) ); // compatibility of inner dimensions
83 assert( _nrows == static_cast<int>(r.size()) ); // compatibility of outer dimensions
84 assert( r.size() == f.size() );
85
86 auto const &D = Matrix::GetDiag(); // accumulated diagonal of matrix @p SK.
87
88 if (zero) { // assumes initial solution is zero
89 for (int k = 0; k < _nrows; ++k) {
90 // u := u + om*D^{-1}*f
91 u[k] = omega*f[k] / D[k]; // MPI: distributed to accumulated vector needed
92 }
93 --nsmooth; // first smoothing sweep done
94 }
95
96
97 for (int ns = 1; ns <= nsmooth; ++ns) {
98 for (int row = 0; row < _nrows; ++row) {
99 double wi = f[row];
100 for (int ij = _id[row]; ij < _id[row + 1]; ++ij) {
101 wi -= _sk[ij] * u[ _ik[ij] ];
102 }
103 r[row] = wi;
104 }
105 for (int k = 0; k < _nrows; ++k) {
106 // u := u + om*D^{-1}*r
107 u[k] = u[k] + omega * r[k] / D[k]; // MPI: distributed to accumulated vector needed
108 }

129int CRS_Matrix::fetch(int const row, int const col) const
130{
131 int const id2 = _id[row + 1]; // end and
132 int ip = _id[row]; // start of recent row (global index)
133
134 while (ip < id2 && _ik[ip] != col) { // find index col (global index)
135 ++ip;
136 }
137 if (ip >= id2) {
138 ip = -1;
139#ifndef NDEBUG // compiler option -DNDEBUG switches off the check
140 cout << "No column " << col << " in row " << row << endl;
141 assert(ip >= id2);
142#endif
143 }
144 return ip;

285 auto tstart = clock();
286 int const nelem(_mesh.Nelems());
287 int const ndof_e(_mesh.NdofsElement());
288 auto const &ia(_mesh.GetConnectivity());

292 assert(*min_element(ia.cbegin(), ia.cbegin() + ndof_e * nelem) == 0); // numbering starts with 0 ?
293
294// Collect for each node those nodes it is connected to (multiple entries)
295// Detect the neighboring nodes
296 vector< list<int> > cc(_nrows); // cc[i] is the list of nodes a node 'i' is connected to
297 for (int i = 0; i < nelem; ++i) {
298 int const idx = ndof_e * i;
299 for (int k = 0; k < ndof_e; ++k) {
300 list<int> &cck = cc[ia[idx + k]];
301 cck.insert( cck.end(), ia.cbegin() + idx, ia.cbegin() + idx + ndof_e );
302 }
303 }
304// Delete the multiple entries
305 _nnz = 0;
306 for (auto &it : cc) {
307 it.sort();
308 it.unique();
309 _nnz += static_cast<int>(it.size());
310 // cout << it.size() << " :: "; copy(it->begin(),it->end(), ostream_iterator<int,char>(cout," ")); cout << endl;
311 }
312
313// CSR data allocation
314 _id.resize(_nrows + 1); // Allocate memory for CSR row pointer
315 _ik.resize(_nnz); // Allocate memory for CSR column index vector
316
317// copy CSR data
318 _id[0] = 0; // begin of first row
319 for (size_t i = 0; i < cc.size(); ++i) {
320 //cout << i << " " << nid.at(i) << endl;;
321 const list<int> &ci = cc[i];
322 const auto nci = static_cast<int>(ci.size());
323 _id[i + 1] = _id[i] + nci; // begin of next line
324 copy(ci.begin(), ci.end(), _ik.begin() + _id[i] );
325 }
326
327 assert(_nnz == _id[_nrows]);
328 _sk.resize(_nnz); // Allocate memory for CSR column index vector
329
330 _ncols = *max_element(_ik.cbegin(), _ik.cend()); // maximal column number
331 ++_ncols; // node numbering: 0 ... nnode-1
332 //cout << _nrows << " " << _ncols << endl;
333 assert(_ncols == _nrows);
334
335 double duration = static_cast<double>(clock() - tstart) / CLOCKS_PER_SEC; // ToDo: change to systemclock

369void FEM_Matrix::ApplyDirichletBC(std::vector<double> const &u, std::vector<double> &f)
370{
371 auto const idx = _mesh.Index_DirichletNodes();
372 int const nidx = static_cast<int>(idx.size());
373
374 for (int i = 0; i < nidx; ++i) {
375 int const row = idx[i];
376 for (int ij = _id[row]; ij < _id[row + 1]; ++ij) {
377 int const col = _ik[ij];
378 if (col == row) {
379 _sk[ij] = 1.0;
380 f[row] = u[row];
381 }
382 else {
383 int const id1 = fetch(col, row); // Find entry (col,row)
384 assert(id1 >= 0);
385 f[col] -= _sk[id1] * u[row];
386 _sk[id1] = 0.0;
387 _sk[ij] = 0.0;
388 }
389 }
390 }

392 return;

563

570 ske[2][0] = ske[0][2];
571 ske[2][1] = ske[1][2];
572 ske[2][2] = 0.5 / jac * (y21 * y21 + x21 * x21);
573
574 const double xm = (xc[i1 + 0] + xc[i2 + 0] + xc[i3 + 0]) / 3.0,

632void BisectInterpolation::Mult(vector<double> &wf, vector<double> const &uc) const
633{
634 assert( Nrows() == static_cast<int>(wf.size()) );
635 assert( Ncols() == static_cast<int>(uc.size()) );
636
637 for (int k = 0; k < Nrows(); ++k) {
638 wf[k] = _vv[2 * k] * uc[_iv[2 * k]] + _vv[2 * k + 1] * uc[_iv[2 * k + 1]];

643void BisectInterpolation::MultT(vector<double> const &wf, vector<double> &uc) const
644{
645 assert( Nrows() == static_cast<int>( wf.size()) );
646 assert( Ncols() == static_cast<int>( uc.size()) );
647 assert(2*Nrows() == static_cast<int>(_iv.size()) );
648 assert(2*Nrows() == static_cast<int>(_vv.size()) );
649
650 for (int k = 0; k < Ncols(); ++k) uc[k] = 0.0;
651
652// GH: atomic slows down the code ==> use different storage for MultT operation (CRS-matrix?)
653 for (int k = 0; k < Nrows(); ++k) {
654 int const j1=_iv[2 * k ];
655 int const j2=_iv[2 * k + 1];
656 uc[j1] += _vv[2 * k ] * wf[k];
657 uc[j2] += _vv[2 * k + 1] * wf[k];
658 }
659 return;
660}

663void BisectInterpolation::MultT_Full(vector<double> const &wf, vector<double> &uc) const
664{
665 assert( Nrows() == static_cast<int>(wf.size()) );
666 assert( Ncols() == static_cast<int>(uc.size()) );
667// GH: atomic slows down the code ==> use different storage for MultT operation (CRS-matrix?)
668 for (int k = 0; k < Ncols(); ++k) uc[k] = 0.0;
669 vector<double> full(uc.size(),0.0);
670 for (int k = 0; k < Nrows(); ++k) {
671 if (_iv[2 * k] != _iv[2 * k + 1]) {
672 uc[_iv[2 * k] ] += _vv[2 * k ] * wf[k];
673 uc[_iv[2 * k + 1]] += _vv[2 * k + 1] * wf[k];
674 full[_iv[2 * k ]] += _vv[2 * k ];
675 full[_iv[2 * k + 1]] += _vv[2 * k + 1];
676 }
677 else {
678 uc[_iv[2 * k] ] += 2.0*_vv[2 * k ] * wf[k]; // uses a property of class BisectInterpolation
679 full[_iv[2 * k] ] += 2.0*_vv[2 * k ];
680 }
681 }
682 for (size_t k=0; k<uc.size(); ++k) uc[k] /= full[k];
683 return;
684}
