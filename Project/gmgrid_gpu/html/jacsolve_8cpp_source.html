249{
250 int lev=Nlevels()-1; // fine level
251
252 double s0(-1);
253 double si(0);
254 // start with zero guess
255 DiagPrecond(_vSK[lev], _f[lev], _u[lev], 1.0); // w := D^{-1]*f
256 dscapr(_f[lev],_u[lev],s0); // s_0 := <f,u>
257
258 bool bzero = true; // start with zero guess
259 int iter = 0;
260 do
261 {
262 MG_Step(lev, pre_smooth, bzero, nu);
263 bzero=false;
264 _vSK[lev].Defect(_d[lev], _f[lev], _u[lev]); // d := f - K*u
265 DiagPrecond(_vSK[lev], _d[lev], _w[lev], 1.0); // w := D^{-1]*d
266 dscapr(_d[lev],_w[lev], si); // s_i := <d,w>
267 ++iter;
268 } while (si>s0*eps*eps && iter<1000);
